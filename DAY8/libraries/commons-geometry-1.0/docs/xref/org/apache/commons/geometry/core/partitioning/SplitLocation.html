17  package org.apache.commons.geometry.core.partitioning;
18  
19  /** Enumeration representing the location of a split object with respect
20   * to its splitting {@link Hyperplane hyperplane}.
21   */
22  public enum SplitLocation {
23  
24      /** Value indicating that the split object lies entirely on the
25       * plus side of the splitting hyperplane.
26       */
27      PLUS,
28  
29      /** Value indicating that the split object lies entirely on the
30       * minus side of the splitting hyperplane.
31       */
32      MINUS,
33  
34      /** Value indicating that the split object lies in both the plus
35       * and minus sides of the splitting hyperplane.
36       */
37      BOTH,
38  
39      /** Value indicating that the split object lies neither on the plus
40       * or minus sides of the splitting hyperplane. This is the case when
41       * the object lies entirely on the hyperplane or is empty (and
42       * therefore "lies" nowhere).
43       */
44      NEITHER
45  }
