17  package org.apache.commons.geometry.core.partitioning.bsp;
18  
19  /** Enum describing the possible behaviors when cutting a region BSP tree node
20   * with a hyperplane to produce two new child nodes.
21   */
22  public enum RegionCutRule {
23  
24      /** Set the minus side of the cutting hyperplane as the inside of the region
25       * and the plus side as the outside. This is the default convention for hyperplanes.
26       */
27      MINUS_INSIDE,
28  
29      /** Set the plus side of the cutting hyperplane as the inside of the region and
30       * the minus side as the outside.
31       */
32      PLUS_INSIDE,
33  
34      /** Set both child nodes to the same location as the parent node. For example, if the
35       * parent node is marked as inside, both child nodes will be marked as inside. Similarly
36       * if the parent node is marked as outside, both child nodes will be marked as outside.
37       * This rule can be used to modify the tree structure (to perhaps produce a more efficient,
38       * balanced tree) without changing the represented region.
39       */
40      INHERIT
41  }
